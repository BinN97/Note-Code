    if(逻辑条件){
        语句快;
    }

    if(逻辑条件){
        语句块1；
    }else{
        语句块2；
    }
    if(逻辑条件1){
        语句块1;
    }else if{逻辑条件2){
        语句块2;
    }...

    }
